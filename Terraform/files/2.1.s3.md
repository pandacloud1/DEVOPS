# CREATE S3 BUCKET USING TERRAFORM

```sh
variables.tf
```
```hcl
# Define input variables
variable "bucket" {
  default = "pandabucket01012020"     # give unique bucket name
}

variable "acl" {
  default = "private"
}

variables "tags" {
  type = map(string)
  default = {
    Name = "My-Bucket",
    Env = "Dev",
    Owner = "PandaCloud"
  }
}
```
```sh
main.tf
```
```hcl
# Create S3 bucket
resource "aws_s3_bucket" "mybucket" {
  bucket = var.bucket
  acl = var.acl
  tags = var.tags
}
```
```sh
outputs.tf
```
```hcl
# S3 bucket name as output
output "S3_bucket_name" {
  value = aws_s3_bucket.mybucket.bucket
}

# S3 acl value output
output "acl_name" {
  value = aws_s3_bucket.mybucket.acl
}

# S3 tags output
output "s3_tags" {
  value = aws_s3_bucket.mybucket.tags
}
```
```sh
terraform apply --auto-approve
```
```sh
terraform destroy --auto-approve
```
